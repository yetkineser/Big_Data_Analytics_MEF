import pylab as pl
from sklearn.metrics import confusion_matrix
from sklearn.tree import DecisionTreeClassifier
from sklearn.datasets import load_breast_cancer
from sklearn.model_selection import train_test_split  # some documents still include the cross-validation option but it no more exists in version 18.0
import pylab as plt

cancer = load_breast_cancer()
X_train, X_test, y_train, y_test = train_test_split(cancer.data, cancer.target, stratify=cancer.target, random_state=42)
k=1
b = []
d = []
e = []
while (k <15):

    tree = DecisionTreeClassifier(max_depth=k, random_state=0)
    fit = tree.fit(X_train, y_train)
    predicted_test = fit.predict(X_test)
    predicted_train = fit.predict(X_train)
    b.append(round(accuracy_score(y_test, predicted_test),4))
    d.append(k)
    e.append(round(accuracy_score(y_train, predicted_train),4))
    print("Max Depth = ",k)
    print(confusion_matrix(y_test,predicted_test))
    k += 1

pl.figure(figsize=(15,8))
pl.plot(d,b,'x-',label = 'Xsaved')
pl.plot(d,e,'x-',label = 'Xsaved')
pl.xlabel('Max Depth')
pl.ylabel('Accuracy')
pl.show()
