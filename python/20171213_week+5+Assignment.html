class RectengularPrism():
    def __init__(self, l, h, w):
        self.length = l    
        self.height = h
        self.width = w
    def area(self):
        return  (2 * self.length * self.height) + (2 * self.length * self.width) + (2 * self.width * self.height)
    def volume(self):
        return (self.length * self.height * self.width)
    
    def PrintCalculated(self):
        print("Area = {0}".format(RectengularPrism.area(self)))
        print("Volume = {0}".format(RectengularPrism.volume(self)))
